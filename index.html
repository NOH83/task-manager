구글 시트 연동 방법
  1. 구글 시트에서 확장 프로그램 → Apps Script 열기
  2. 아래 코드 붙여넣고 저장 후 웹 앱으로 배포 (액세스: 모든 사용자)
  3. 배포 URL을 '구글 시트 연동' 버튼에 입력
function doPost(e) { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.getActiveSpreadsheet() .getActiveSheet(); if (data.action === 'sync') { sheet.clearContents(); sheet.appendRow(['업무명','마감일','완료여부']); data.tasks.forEach(t => sheet.appendRow([t.name, t.due, t.done ? '완료' : '진행중']) ); } return ContentService.createTextOutput('OK'); }
